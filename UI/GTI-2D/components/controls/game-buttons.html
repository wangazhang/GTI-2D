使用示例

// 创建游戏按钮实例
const fireButton = new GameButton('fire-button', {
    type: 'fire',
    size: 'large',
    icon: '🔫',
    label: '射击',
    cooldown: 0,
    autoRepeat: true,
    repeatInterval: 100,
    hapticFeedback: true,
    soundEffect: 'shoot.wav'
});

// 创建技能按钮
const skillButton = new GameButton('skill-button', {
    type: 'skill',
    size: 'medium',
    icon: '⚡',
    label: '雷击',
    cooldown: 8000,
    energyCost: 50,
    maxEnergy: 100,
    currentEnergy: 100
});

// 创建长按按钮
const holdButton = new GameButton('hold-button', {
    type: 'hold',
    size: 'large',
    icon: '🔥',
    label: '充能攻击',
    holdTime: 3000,
    showProgress: true
});

// 监听事件
fireButton.on('press', () => {
    console.log('开火！');
    // 执行射击逻辑
});

skillButton.on('press', () => {
    if (skillButton.canUse()) {
        console.log('释放技能');
        skillButton.startCooldown();
        skillButton.consumeEnergy(50);
    }
});

holdButton.on('hold', (progress) => {
    if (progress >= 100) {
        console.log('充能完成，释放强力攻击！');
    }
});

// 更新按钮状态
skillButton.setEnergy(75);
skillButton.setCooldown(5000);

// 连击系统
fireButton.on('combo', (count) => {
    console.log('连击数:', count);
    if (count >= 10) {
        console.log('连击奖励！');
    }
});

// 启用/禁用按钮
skillButton.disable();
skillButton.enable();

// 设置按钮样式
fireButton.setStyle('danger');
skillButton.setGlow(true);
