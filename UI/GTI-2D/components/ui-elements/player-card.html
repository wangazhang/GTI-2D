使用示例

// 创建玩家卡片实例
const playerCard = new PlayerCard('container-id', {
    id: 'player_001',
    name: 'ShadowStrike',
    avatar: '🔫',
    level: 25,
    rank: 'Elite',
    status: 'online',        // online, away, offline, busy
    team: 'alpha',          // alpha, bravo, neutral
    health: 85,             // 0-100
    ping: 42,               // ms
    kills: 15,
    deaths: 3,
    assists: 8,
    score: 2450,
    loadout: ['🔫', '🔪', '💊'], // 装备图标
    showActions: true,      // 显示操作按钮
    showStats: true,        // 显示战绩
    showHealth: true,       // 显示血量
    size: 'standard',       // compact, standard, large
    selectable: true,       // 可选择
    draggable: false        // 可拖拽
});

// 监听事件
playerCard.on('click', (player) => {
    console.log('玩家被点击:', player);
});

playerCard.on('action', ({ action, player }) => {
    switch(action) {
        case 'invite':
            console.log('邀请玩家:', player.name);
            break;
        case 'kick':
            console.log('踢出玩家:', player.name);
            break;
        case 'mute':
            console.log('禁言玩家:', player.name);
            break;
        case 'profile':
            console.log('查看资料:', player.name);
            break;
    }
});

// 更新玩家状态
playerCard.setStatus('away');
playerCard.setHealth(60);
playerCard.setPing(85);
playerCard.updateStats({
    kills: 18,
    deaths: 4,
    score: 2850
});

// 显示语音指示
playerCard.showVoiceActivity();

// 显示打字指示
playerCard.showTyping();

// 选中玩家
playerCard.select();

// 设置队伍
playerCard.setTeam('bravo');
